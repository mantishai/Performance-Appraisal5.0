## 一、功能概述

在个人看板页面"个人信息卡片"中，点击"✏️ 编辑个人信息"按钮，弹出模态弹窗，用于查看和编辑当前登录用户的个人详细信息。

弹窗包含以下内容模块：
1. 基本信息 + 一寸照片上传
2. 身份信息
3. 联系信息
4. 学历与职称信息
5. 工作信息（部门-岗位联动）
6. 薪酬福利
7. 附件
8. 详细信息表格（11个子表格）

async function openPersonalInfoModal() {
    // 1. 检查是否登录

    }

}

async function updatePositionOptions() {
    const department = document.getElementById('modalDepartment').value;
    const positionSelect = document.getElementById('modalPosition');
    
    positionSelect.innerHTML = '<option value="">请选择工作岗位</option>';
    if (!department) return;
    
    try {
        // 从数据库API加载该部门的岗位列表
        const res = await fetch(`/api/positions?department=${encodeURIComponent(department)}`, {
            headers: { 'x-session-id': sessionId }
        });
        if (res.ok) {
            const data = await res.json();
            data.forEach(pos => {
                const option = document.createElement('option');
                option.value = pos.name;
                option.textContent = pos.name;
                positionSelect.appendChild(option);
            });
        }
    } catch (e) {
        console.error('加载岗位列表失败:', e);
    }
}

async function savePersonalInfo() {
    const name = document.getElementById('modalName').value;
    if (!name.trim()) {
        showToast('请输入姓名', 'warning');
        return;
    }
    
    // 收集所有表单数据
    const personalInfo = {
        photo: personalPhoto,
        name: name,
        gender: document.getElementById('modalGender').value,
        // ... 所有字段
        workExperiences: collectWorkExperiences(),
        familyMembers: collectFamilyMembers(),
        // ... 其他子表格数据
    };
    
    try {
        const res = await fetch(`/api/user/${currentUser.id}`, {
            method: 'PUT',
            headers: {
                'Content-Type': 'application/json',
                'x-session-id': sessionId
            },
            body: JSON.stringify(personalInfo)
        });
        
        if (res.ok) {
            // 更新本地 users 数组中的用户数据
            const userIndex = users.findIndex(u => u.id === currentUser.id);
            if (userIndex !== -1) {
                users[userIndex] = { ...users[userIndex], ...personalInfo };
            }
            
            updatePersonalInfoDisplay(personalInfo);
            closePersonalInfoModal();
            showToast('✅ 个人信息已保存', 'success');
        }
    } catch (e) {
        showToast('⚠️ 保存失败，请重试', 'error');
    }
}

function collectWorkExperiences() {

function exportPersonalInfo() {
    const user = users.find(u => u.id === currentUser.id);
    if (!user) { showToast('暂无数据', 'warning'); return; }
    
    const blob = new Blob([JSON.stringify(user, null, 2)], { type: 'application/json' });
    const link = document.createElement('a');
    link.href = URL.createObjectURL(blob);
    link.download = `个人信息_${user.name}_${new Date().toISOString().slice(0,10)}.json`;
    link.click();
    showToast('✅ 导出成功', 'success');
}

function importPersonalInfo() {
    const input = document.createElement('input');
    input.type = 'file';
    input.accept = '.json';
    input.onchange = async function(e) {
        const file = e.target.files[0];
        const text = await file.text();
        try {
            const data = JSON.parse(text);
            loadPersonalInfoToModal(data);
            showToast('✅ 导入成功，请点击保存确认', 'success');
        } catch (e) {
            showToast('❌ 文件格式不正确', 'error');
        }
    };
    input.click();
}
六、数据库相关（参考 server/ 和 sql/ 目录）
用户表（示例）：

users 表包含所有基本信息字段

work_experiences 表关联 user_id

family_members 表关联 user_id

其他子表格同样关联 user_id

API端点：

GET /api/user/:id - 获取用户完整信息（含关联子表数据）

PUT /api/user/:id - 更新用户完整信息（建议使用事务保存主表+子表）

GET /api/departments - 获取部门列表

GET /api/positions?department=xxx - 获取某部门下的岗位列表

七、注意事项
数据来源：用户数据从数据库API加载，不要使用 localStorage 作为主要存储

字段映射：注意数据库字段名（如 direct_superior）与前端变量名（如 directSuperior）的转换

部门-岗位联动：选择部门后，从 /api/positions?department=xxx 加载岗位列表

照片存储：照片使用 base64 存储在数据库的 photo 字段（MEDIUMTEXT类型）

手机号脱敏：只在看板展示时脱敏，弹窗中显示完整号码

子表格：建议后端使用事务保存主表+11个子表，确保数据一致性

错误处理：所有API调用需try-catch，失败时给出友好提示
